include <string>
using namespace std;
int main() {
string input;
regex integer{"[+-]?\\d*\\.?\\d+"};
cout << "Enter some integers:" << endl;
int sum = 0;
while(cin>>input) {
if(regex_match(input,integer))
sum += stoi(input);
else
cerr << "Error: Not an integer" << endl;
}
cout << "Sum is " << sum << endl;
}
